sequenceDiagram
    autonumber
    participant Shipper as Verlader (Frontend)
    participant API as Backend API
    participant Match as Matching Service
    participant ML as ML-Service
    participant Wallet as Wallet-Service
    participant Stripe as Stripe
    participant Transporter as Transporteur (Frontend)

    rect rgb(239, 246, 255)
    Note over Shipper,API: Phase 1: Job Creation
    Shipper->>API: POST /jobs (Neuen Auftrag erstellen)
    API->>API: Job in DB anlegen (status = "open")
    API-->>Shipper: Job-Details
    end

    rect rgb(240, 253, 244)
    Note over Match,ML: Phase 2: Matching
    API->>Match: match_transporters_for_job(job)
    Match->>API: Kandidaten aus DB laden
    loop pro Kandidat
        Match->>ML: POST /score (Features)
        ML-->>Match: scoreMl
    end
    Match-->>API: Top-N Transporteur-IDs (mit Score)
    API->>Transporter: Notify (In-App / E-Mail: Neuer Job)
    end

    rect rgb(255, 247, 237)
    Note over Transporter,API: Phase 3: Bidding
    Transporter->>API: POST /jobs/{id}/bids (Angebot abgeben)
    API->>API: Bid speichern (status = "open")
    API-->>Transporter: Bid bestätigt
    Shipper->>API: GET /jobs/{id}/bids (Angebote ansehen)
    end

    rect rgb(245, 243, 255)
    Note over Shipper,Wallet: Phase 4: Accept Bid & Wallet Booking
    Shipper->>API: PATCH /jobs/{id}/accept_bid (bid_id)
    API->>API: Job & Bid validieren (Status, Ownership)
    API->>Wallet: get_wallet_by_user(shipper)
    Wallet-->>API: Wallet-Daten

    alt Wallet-Balance < Bid-Preis
        API-->>Shipper: 400 Insufficient wallet balance
    else Wallet-Balance ok
        API->>Wallet: debit_wallet(shipper, amount, "booking")
        API->>Wallet: credit_wallet(platform, fee)
        API->>Wallet: credit_wallet(transporter, amount-fee)
        API->>API: Job.status = "booked", Bid.status = "accepted"
        API->>API: andere Bids = "rejected"
        API-->>Shipper: Booking bestätigt
        API-->>Transporter: Du wurdest gebucht
    end
    end

    rect rgb(254, 243, 199)
    Note over Transporter,API: Phase 5: Transport Execution
    Transporter->>API: Status-Updates (in_progress, completed)
    API->>API: Job.status aktualisieren
    end

    rect rgb(254, 226, 226)
    Note over API,Stripe: Phase 6: Payout
    API->>API: (Worker/Cron) findet completed + payout_pending
    API->>Wallet: check transporter wallet balance
    API->>Stripe: Transfer.create(...) (Stripe Connect Payout)
    Stripe-->>API: Transfer bestätigt
    API->>Wallet: debit_wallet(transporter, amount, "payout")
    API->>API: Job.payout_status = "paid"
    end
