    rect rgb(245, 243, 255)
    Note over ML,Model: Phase 4: Model Training & Registry
    
    ML->>ML: train XGBoost/LightGBM on features
    ML->>ML: evaluate (AUC, Precision@K, NDCG)
    ML->>ML: cross_validate (5-fold)
    
    alt Model improved (AUC > baseline)
        ML->>Model: register_model(version, metrics, feature_importance)
        Model-->>ML: model_id
        ML->>API: notify_new_model_available(model_id)
    else No improvement
        ML->>ML: log experiment, discard model
    end
    end

    rect rgb(254, 243, 199)
    Note over API,Model: Phase 5: Online Serving (Inference)
    
    API->>Store: GET transporter_features(transporter_id)
    Store-->>API: {rating, cancel_rate, jobs_count, ...}
    
    API->>API: build_features(job, transporter)
    API->>ML: POST /score (features)
    ML->>Model: load_latest_model()
    Model-->>ML: model_weights
    ML->>ML: predict(features) → scoreMl
    ML-->>API: {score: 0.87, confidence: 0.92}
    
    Note right of API: Ranking: S = 0.5 × H + 0.5 × M
    API->>API: combine_heuristic_ml(heuristic, ml_score)
    API-->>API: ranked_transporters[]
    end

    rect rgb(254, 226, 226)
    Note over ML,Model: Phase 6: Feedback Loop
    
    API->>Events: publish("prediction_made", {job_id, transporter_id, score, actual_outcome?})
    
    loop Weekly Retraining
        ML->>Store: SELECT predictions WHERE created_at > 7_days_ago
        Store-->>ML: recent_predictions[]
        ML->>ML: compute prediction_accuracy (calibration)
        ML->>ML: detect feature_drift (KS-test)
        
        alt Drift detected OR accuracy dropped
            ML->>ML: trigger_retraining()
        end
    end
