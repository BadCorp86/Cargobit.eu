sequenceDiagram
    autonumber
    participant TApp as Transporteur App
    participant API as Backend API
    participant DB as Database
    participant WS as WebSocket Server
    participant SApp as Shipper App
    participant Notify as Notification Service

    rect rgb(239, 246, 255)
    Note over TApp,DB: Phase 1: Location Updates (Periodic)
    loop alle 30 Sekunden
        TApp->>API: POST /jobs/{id}/location
{lat, lng, speed, heading}
        API->>DB: UPDATE job_locations
        API->>WS: broadcast("location_update", job_id, location)
        WS-->>SApp: {lat, lng, timestamp}
    end
    end

    rect rgb(240, 253, 244)
    Note over TApp,DB: Phase 2: Status Transitions
    TApp->>API: PATCH /jobs/{id}/status
{status: "in_transit"}
    API->>DB: UPDATE jobs SET status = "in_transit"
    API->>DB: INSERT status_history
    API->>WS: broadcast("status_change", job_id, "in_transit")
    WS-->>SApp: Job gestartet
    API->>Notify: push_notification(shipper_id, "Transport gestartet")
    Notify-->>SApp: 🔔 Push Notification
    end

    rect rgb(255, 247, 237)
    Note over TApp,DB: Phase 3: Checkpoints & Milestones
    TApp->>API: POST /jobs/{id}/checkpoints
{type: "pickup", location, photos[]}
    API->>DB: INSERT checkpoints
    API->>DB: UPDATE jobs SET pickup_at = NOW()
    API->>WS: broadcast("checkpoint", job_id, pickup_data)
    WS-->>SApp: Abholung bestätigt
    Note over TApp,API: Optional: Signature Capture
    TApp->>API: POST /jobs/{id}/signature
{signature_base64, recipient_name}
    API->>DB: UPDATE jobs SET signature_url
    end

    rect rgb(245, 243, 255)
    Note over TApp,Notify: Phase 4: Delivery Confirmation
    TApp->>API: PATCH /jobs/{id}/status
{status: "delivered", delivery_proof}
    API->>DB: UPDATE jobs SET status = "delivered"
    API->>DB: UPDATE jobs SET delivered_at = NOW()
    API->>WS: broadcast("status_change", job_id, "delivered")
    WS-->>SApp: ✅ Lieferung abgeschlossen
    API->>Notify: push_notification(shipper_id, "Lieferung erfolgt")
    API->>Notify: email_receipt(shipper_id, job_details)
    end

    rect rgb(254, 243, 199)
    Note over SApp,DB: Phase 5: History & Audit Trail
    SApp->>API: GET /jobs/{id}/timeline
    API->>DB: SELECT * FROM status_history WHERE job_id
    DB-->>API: timeline_events[]
    API-->>SApp: [{status, timestamp, location, actor}]
    SApp->>API: GET /jobs/{id}/tracking
    API->>DB: SELECT * FROM job_locations ORDER BY timestamp
    DB-->>API: location_history[]
    API-->>SApp: Polyline für Karte
    end
