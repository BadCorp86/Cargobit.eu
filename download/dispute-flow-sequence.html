sequenceDiagram
    autonumber
    participant Shipper as Verlader
    participant Transporter as Transporteur
    participant API as Backend API
    participant Wallet as Wallet Service
    participant Stripe as Stripe
    participant Admin as Admin Panel
    participant DB as Database

    rect rgb(254, 226, 226)
    Note over Shipper,DB: Phase 1: Storno (vor Pickup)
    Shipper->>API: POST /jobs/{id}/cancel
{reason: "no_transporter_available"}
    API->>DB: CHECK job.status IN ("open", "booked")
    
    alt Job Status ok für Storno
        API->>DB: UPDATE job SET status = "cancelled"
        API->>DB: INSERT dispute_record
        
        alt Refund erforderlich (booked)
            API->>Wallet: get_wallet_by_user(shipper)
            API->>Wallet: credit_wallet(shipper, full_amount, "refund")
            API->>Wallet: debit_wallet(transporter, pending_amount, "cancel")
            API->>Wallet: debit_wallet(platform, fee, "cancel_fee_return")
        end
        
        API->>Transporter: Notify: Job storniert
        API-->>Shipper: 200 Storno bestätigt
    else Job bereits in_transit
        API-->>Shipper: 400 Cannot cancel - transport in progress
    end
    end

    rect rgb(254, 243, 199)
    Note over Shipper,DB: Phase 2: Refund (nach Delivery - Mängel)
    Shipper->>API: POST /disputes
{job_id, type: "damage", evidence[]}
    API->>DB: INSERT disputes (status = "open")
    API->>Admin: Notify: Neuer Dispute
    API-->>Shipper: Dispute eingereicht

    Admin->>API: GET /disputes/{id}
    API-->>Admin: Dispute Details + Evidence
    
    Admin->>API: PATCH /disputes/{id}/resolve
{action: "partial_refund", amount_cents}
    
    alt Partial Refund
        API->>Stripe: Refund.create({amount, payment_intent})
        Stripe-->>API: Refund confirmed
        API->>Wallet: credit_wallet(shipper, refund_amount, "dispute_refund")
        API->>DB: UPDATE disputes SET status = "resolved"
        API-->>Shipper: Refund verarbeitet
        API-->>Transporter: Dispute resolved - partial refund
    else Full Refund
        API->>Stripe: Refund.create({amount: full_amount})
        Stripe-->>API: Refund confirmed
        API->>Wallet: credit_wallet(shipper, full_amount, "full_refund")
        API->>Wallet: debit_wallet(transporter, payout, "chargeback")
        API->>DB: UPDATE disputes SET status = "resolved"
    else Rejected
        API->>DB: UPDATE disputes SET status = "rejected"
        API-->>Shipper: Dispute abgelehnt (Begründung)
    end
    end

    rect rgb(245, 243, 255)
    Note over Shipper,DB: Phase 3: Eskalation (Admin-Entscheid)
    Shipper->>API: POST /disputes/{id}/escalate
{additional_evidence}
    API->>DB: UPDATE disputes SET escalated = true, status = "escalated"
    API->>Admin: High-Priority Alert: Eskalation

    Admin->>API: PATCH /disputes/{id}/final_decision
{outcome, notes, adjust_rating?}
    
    alt Outcome: Shipper favor
        API->>Stripe: Refund.create(...)
        API->>Wallet: credit_wallet(shipper, amount)
        API->>DB: UPDATE users SET rating = rating - 0.2 WHERE transporter
        API-->>Shipper: ✅ Entscheidung zu Ihren Gunsten
        API-->>Transporter: ⚠️ Rating angepasst
    else Outcome: Transporter favor
        API->>DB: UPDATE disputes SET status = "closed"
        API-->>Shipper: ❌ Dispute abgelehnt
        API-->>Transporter: ✅ Dispute geschlossen - keine Aktion
    else Outcome: Split
        API->>Wallet: credit_wallet(shipper, partial)
        API-->>Shipper: Teilweise Rückerstattung
        API-->>Transporter: Teilweise belastet
    end
    
    API->>DB: INSERT audit_log (decision, actor, timestamp)
    end
